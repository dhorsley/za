[]string = argv()

top

CLI arguments as an array.

Example:

# simple getopt

define next()
    lav=getglob("av")
    arg=lav.head
    lav=lav.tail
    setglob av = lav
    return arg
end    

av=argv()

while !av.empty
    arg=next()
    if arg.len>1 && start(arg,"-")
        opt=substr(arg,0,2)
        case opt
        is "-h"
            println "Help requested."
        is "-s"
            svar=next().string
        is "-b"
            bvar=next().bool
        is "-n"
            nvar=next().int
        endcase
    else
        println "unprocessed argument found: {arg}"
    endif
endwhile

println "string option : {svar} [{=kind(svar)}]"
println "bool option   : {bvar} [{=kind(bvar)}]"
println "number option : {nvar} [{=kind(nvar)}]"
